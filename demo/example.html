            /**
             * When this page tour ends, 
             * Angular Tour will emit the onTourEnd event
             * so you can control the navigation
             * and demonstrate the awesomenuess of your app!
             */
            app.controller('MyCtrl', function($scope, $location) {
              $scope.init = function() {
                  $scope.$on('onTourEnd', function() {
                     $location.path('/nextpage);
                  });
               }
             })
